```sql
CREATE TABLE orders (
    id SERIAL PRIMARY KEY,
    customer_name TEXT,
    amount NUMERIC,
    created_at TIMESTAMP DEFAULT now()
);

CREATE TABLE order_log (
    id SERIAL PRIMARY KEY,
    order_id INT,
    logged_at TIMESTAMP DEFAULT now()
);

-- без этой функции триггер работать не будет
CREATE OR REPLACE FUNCTION log_order_insert()
RETURNS TRIGGER AS $$
BEGIN
    INSERT INTO order_log(order_id, logged_at)
    VALUES (NEW.id, now());
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trigger_log_order
AFTER INSERT ON orders
FOR EACH ROW
EXECUTE FUNCTION log_order_insert();

INSERT INTO orders(customer_name, amount)
VALUES ('Alice', 120.50);
```
