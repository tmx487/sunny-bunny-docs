```csharp
// A file is being flushed before opening
// StreamWriter - блокирует файл на время своей работы, поэтому не получится из другого StreamWriter писать в зайдествованный файл
var textWriter = new StreamWriter("text.txt");

textWriter.WriteLine("String #1");
textWriter.WriteLine("String #2");

textWriter.Flush(); // осуществляет принудительный сброс буфера на диск, т.е. произойдет запись данных на диск
textWriter.WriteLine("String #3");
textWriter.Close();

var textWriter = new StreamWriter("text.txt");
while(true)
{
	var str = textWriter.ReadLine();
	if(str == null) break;
	Console.WriteLine(str);
}
textWriter.Close();
```
